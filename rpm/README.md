./opt/pacbio/pa-ws-0.0.0/bin/pa-ws
./opt/pacbio/pa-ws-0.0.0/bin/precheck-pa-ws.sh
./opt/pacbio/pa-ws-0.0.0/systemd/pacbio-pa-ws-0.0.0.service
./opt/pacbio/pa-ws-0.0.0/systemd/pacbio-pa-ws.conf
